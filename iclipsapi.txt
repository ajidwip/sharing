    Function IsExistAgent(ByVal pCompanyID As String, ByVal pPolicyID As String, ByVal pAgentID As String, ByRef pmsg As String) As Boolean
        Dim odal As New SQLAccess("iclipscs")
        Dim ord As SqlClient.SqlDataReader
        Try
            pmsg = ""
            ord = odal.ExecuteDataReader("usp_get_records", CommandType.StoredProcedure, _
                                odal.CreateParameter("@CompanyID", SqlDbType.VarChar, pCompanyID), _
                                odal.CreateParameter("@table", SqlDbType.VarChar, "exist_agent"), _
                                odal.CreateParameter("@skey1", SqlDbType.VarChar, pPolicyID), _
                                odal.CreateParameter("@skey2", SqlDbType.VarChar, pAgentID))
            If ord.HasRows Then
                pmsg += "Agent ID already exist"
            End If

            If pmsg = "" Then
                ord = odal.ExecuteDataReader("usp_get_records", CommandType.StoredProcedure, _
                                odal.CreateParameter("@CompanyID", SqlDbType.VarChar, pCompanyID), _
                                odal.CreateParameter("@table", SqlDbType.VarChar, "exist_agent_name"), _
                                odal.CreateParameter("@skey1", SqlDbType.VarChar, pPolicyID), _
                                odal.CreateParameter("@skey2", SqlDbType.VarChar, pAgentID))
                If ord.HasRows Then
                    pmsg += "Agent name already exist"
                End If
            End If
            
        Catch ex As Exception
            Throw ex
        Finally
            odal.CloseAllConnection()
            odal = Nothing
        End Try
        Return (pmsg = "")
    End Function

    Function Run_Process(ByVal pProc As cProcess, ByVal loProcList As List(Of cProcessList)) As cProcess
        Dim odal As New SQLAccess("iclipscs")
        Dim rdr As IDataReader
        Dim oProc As cProcess
        Try
            rdr = odal.ExecuteDataReader("usp_process_run", CommandType.StoredProcedure,
                                  odal.CreateParameter("@CompanyID", SqlDbType.VarChar, pProc.CompanyID),
                                  odal.CreateParameter("@PolicyID", SqlDbType.VarChar, pProc.PolicyID),
                                  odal.CreateParameter("@ProcessType", SqlDbType.VarChar, pProc.ProcessType),
                                  odal.CreateParameter("@StartPeriode", SqlDbType.Date, pProc.StartPeriode),
                                  odal.CreateParameter("@EndPeriode", SqlDbType.Date, pProc.EndPeriode),
                                  odal.CreateParameter("@ProcessID", SqlDbType.VarChar, pProc.ProcessID),
                                  odal.CreateParameter("@Notes", SqlDbType.VarChar, pProc.Notes),
                                  odal.CreateParameter("@CurrencyID", SqlDbType.VarChar, pProc.CurrencyID),
                                  odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, pProc.ProcessBy),
                                  odal.CreateParameter("@InvNo", SqlDbType.VarChar, pProc.InvoiceNo),
                                  odal.CreateParameter("@ProcessDate", SqlDbType.Date, pProc.ProcessDate),
                                  odal.CreateParameter("@BnkAccCd", SqlDbType.VarChar, pProc.BankAccountCode))
            oProc = New cProcess
            If rdr.Read Then
                oProc.ProcessID = common.GetInteger(rdr.Item("ProcessID"))
                oProc.ErrorMessage = common.GetText(rdr.Item("ErrorMessage"))

                If oProc.ErrorMessage = "" And Left(pProc.ProcessType, 3) = "PAY" Then
                    For Each oPrevPol As cProcessList In loProcList
                        odal.ExecuteCommandSP("usp_save_tpayment_detail", _
                             odal.CreateParameter("@CompanyID", SqlDbType.VarChar, pProc.CompanyID), _
                             odal.CreateParameter("@CertificateID", SqlDbType.VarChar, oPrevPol.CertificateID), _
                             odal.CreateParameter("@ProcessID", SqlDbType.Int, oProc.ProcessID), _
                             odal.CreateParameter("@PlanID", SqlDbType.VarChar, oPrevPol.PlanID), _
                             odal.CreateParameter("@DueDate", SqlDbType.Date, oPrevPol.DueDate), _
                             odal.CreateParameter("@PaymentAmount", SqlDbType.Float, oPrevPol.PaymentAmount), _
                             odal.CreateParameter("@BillAmount", SqlDbType.Float, oPrevPol.BillAmount), _
                             odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, pProc.ProcessBy))

    Function SaveReport_Info(ByVal pmode As String, _
                             ByRef orpt As cTransReport, _
                             ByVal loinfo As List(Of cReportInfo), _
                             ByVal pprocesssby As String) As Boolean
        Dim odal As New SQLAccess("iclipscs")
        Try
            odal.BeginTransactions()
            odal.ExecuteCommandSP("usp_save_ttrans_report", _
                                 odal.CreateParameter("@SaveType", SqlDbType.VarChar, Left(pmode, 1)), _
                                 odal.CreateParameter("@TransReportID", SqlDbType.Int, orpt.TransReportID, 15, ParameterDirection.InputOutput), _
                                 odal.CreateParameter("@ReferenceID", SqlDbType.VarChar, orpt.ReferenceID), _
                                 odal.CreateParameter("@CompanyID", SqlDbType.VarChar, orpt.CompanyID), _
                                 odal.CreateParameter("@ReportID", SqlDbType.VarChar, orpt.ReportID), _
                                 odal.CreateParameter("@ReportLink", SqlDbType.VarChar, orpt.ReportLink), _
                                 odal.CreateParameter("@ReportDate", SqlDbType.DateTime, orpt.ReportDate), _
                                 odal.CreateParameter("@Notes", SqlDbType.VarChar, orpt.Notes), _
                                 odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, pprocesssby))

            If pmode = "N" Then
                orpt.TransReportID = common.GetInteger(odal.ReturnValue(1))
            End If

            'delete existing info
            odal.ExecuteCommandSP("usp_delete_records", _
                             odal.CreateParameter("@table", SqlDbType.VarChar, "key_trans_report_info"), _
                             odal.CreateParameter("@nkey1", SqlDbType.Int, orpt.TransReportID), _
                             odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, pprocesssby))

            'underwriting requirement
            For Each oinfo As cReportInfo In loinfo
                odal.ExecuteCommandSP("usp_save_ttrans_report_info", _
                                 odal.CreateParameter("@SaveType", SqlDbType.VarChar, "N"), _
                                 odal.CreateParameter("@TransReportID", SqlDbType.Int, orpt.TransReportID), _
                                 odal.CreateParameter("@FieldID", SqlDbType.VarChar, oinfo.FieldID), _
                                 odal.CreateParameter("@FieldValue", SqlDbType.VarChar, oinfo.FieldValue), _
                                 odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, pprocesssby))
            Next


            odal.CommitTransactions()
            Return True
        Catch ex As Exception
            If Not IsNothing(odal) Then odal.RollBackTransactions()
            Throw ex
        Finally
            odal.CloseAllConnection()
            odal = Nothing
        End Try
    End Function

                                 odal.CreateParameter("@TaskType", SqlDbType.VarChar, "SRAR"), _
                                 odal.CreateParameter("@SurrenderID", SqlDbType.Int, pSurrenderID), _
                                 odal.CreateParameter("@TaskDate", SqlDbType.DateTime, pApproveDate), _
                                 odal.CreateParameter("@Remark", SqlDbType.VarChar, pRemark), _
                                 odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, pprocesssby))
            odal.CommitTransactions()
            Return True
        Catch ex As Exception
            If Not IsNothing(odal) Then odal.RollBackTransactions()
            Throw ex
        Finally
            odal.CloseAllConnection()
            odal = Nothing
        End Try
    End Function

    Function UploadFile(ByVal pCompanyID As String, ByVal pPolicyID As String, _
                        ByVal pCriteria As String) As cUpload
        Dim odal As New SQLAccess("iclipscs")
        Dim rdr As IDataReader
        Dim oUpload As cUpload
        Dim oany As cAnyParams
        Dim arrfield() As String

        Try
            oany = New cAnyParams

            If pCriteria.ToString.Length > 0 Then
                arrfield = Split(pCriteria, "|")
                oany.skey2 = arrfield(0).ToString   'fileID
                oany.skey3 = arrfield(1).ToString   'EximID
                oany.skey4 = arrfield(2).ToString   'userID
                oany.skey5 = arrfield(3).ToString   'MsgDesc
                oany.dkey1 = arrfield(4) 'App Received Date
            End If

            rdr = odal.ExecuteDataReader("usp_Upload_Certificate", CommandType.StoredProcedure, _
                    odal.CreateParameter("@CompanyID", SqlDbType.VarChar, pCompanyID), _
                    odal.CreateParameter("@PolicyID", SqlDbType.VarChar, pPolicyID), _
                    odal.CreateParameter("@FileID", SqlDbType.VarChar, oany.skey2), _
                    odal.CreateParameter("@EximID", SqlDbType.VarChar, oany.skey3), _
                    odal.CreateParameter("@ProcessBy", SqlDbType.VarChar, oany.skey4), _
                    odal.CreateParameter("@ReturnMsg", SqlDbType.VarChar, oany.skey5), _
                    odal.CreateParameter("@AppReceivedDate", SqlDbType.DateTime, oany.dkey1))
            oUpload = New cUpload
            If rdr.Read Then
                oUpload = New cUpload
                oUpload.MsgID = Common.GetText(rdr.Item("MsgID"))
                oUpload.MsgDesc = Common.GetText(rdr.Item("MsgDesc"))
            End If
        Catch ex As Exception
            Throw ex
        Finally
            odal.CloseAllConnection()
            odal = Nothing
        End Try
        Return oUpload
    End Function

    Function GetListBillDetail(ByVal pCompanyID As String, ByVal pPolicyID As String, ByVal pInvoiceNo As String) As List(Of cBillDetail)
        Dim odal As New SQLAccess("iclipscs")
        Dim rdr As IDataReader
        Dim oBill As cBillDetail
        Dim loBill As List(Of cBillDetail)
        Try
            rdr = odal.ExecuteDataReader("usp_get_records", CommandType.StoredProcedure, _
                                  odal.CreateParameter("@CompanyID", SqlDbType.VarChar, pCompanyID), _
                                  odal.CreateParameter("@table", SqlDbType.VarChar, "list_bill_detail"), _
                                  odal.CreateParameter("@skey1", SqlDbType.VarChar, pPolicyID), _
                                  odal.CreateParameter("@skey2", SqlDbType.VarChar, pInvoiceNo))
            loBill = New List(Of cBillDetail)
            While rdr.Read
                oBill = New cBillDetail
                oBill.ID = common.GetText(rdr.Item("BillID"))
                oBill.InvoiceNo = common.GetText(rdr.Item("InvoiceNo"))
                oBill.CertificateID = common.GetText(rdr.Item("CertificateID"))
                oBill.CertificateName = common.GetText(rdr.Item("CertificateInsured"))
                oBill.CertificateStatusName = common.GetText(rdr.Item("CertificateStatusName"))
                oBill.DueDate = common.CToDate(rdr.Item("DueDate"))
                oBill.PlanID = common.GetText(rdr.Item("PlanID"))
                oBill.BillAmount = common.GetDouble(rdr.Item("BillingAmount"))
                oBill.PaymentAmount = common.GetDouble(rdr.Item("BillingAmount"))
                oBill.PaymentStatus = common.GetText(rdr.Item("PaymentStatus"))
                loBill.Add(oBill)
            End While

        Catch ex As Exception
            Throw ex
        Finally
            rdr = Nothing
            odal.CloseAllConnection()
            odal = Nothing
        End Try
        Return loBill
    End Function
